Create the tables :

CREATE TABLE habits (
    id SERIAL PRIMARY KEY,
    progress INT,
    level INT,
    icon VARCHAR(255)
);


-- Create timetable table
CREATE TABLE timetable (
    id SERIAL PRIMARY KEY,
    time VARCHAR(50),
    activity TEXT,
    details VARCHAR(500)
);


-- Create quotes table
CREATE TABLE quotes (
    id SERIAL PRIMARY KEY,
    text TEXT,
    author VARCHAR(100)
);

-- Create discussions table
CREATE TABLE discussions (
    id SERIAL PRIMARY KEY,
    text TEXT,
    icons VARCHAR(100)
);

CREATE TABLE ingredients (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    type TEXT NOT NULL,
    quantity TEXT NOT NULL,
    expiry_date DATE,
    location TEXT NOT NULL,
    icons VARCHAR(100)
);
