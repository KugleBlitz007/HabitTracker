Create the tables :

CREATE TABLE habits (
    id SERIAL PRIMARY KEY,
    progress INT,
    level INT
);


-- Create timetable table
CREATE TABLE timetable (
    id SERIAL PRIMARY KEY,
    time VARCHAR(50),
    activity TEXT
);


-- Create quotes table
CREATE TABLE quotes (
    id SERIAL PRIMARY KEY,
    text TEXT,
    author VARCHAR(100)
);
